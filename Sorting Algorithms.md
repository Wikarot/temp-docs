## Bubble

<!-- ```JavaScript
let arr = [6, 3, 9, 0];
let n = arr.length;

function bubble() {
  let i, j, swapped;
  for (i = 0; i < n - 1; i++) {
    swapped = false;
    for (j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
        swapped = true;
      }
    }
    if (!swapped) break;
  }
}

console.log(arr);
